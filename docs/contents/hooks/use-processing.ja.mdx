## インポート

```ts
import { useProcessing } from "@yamada-ui/react"
```

## 使い方

```tsx functional=true
const processing = useProcessing()

const onClick = () => {
  processing.start()

  setTimeout(() => {
    processing.finish()
  }, 3000)
}

return (
  <Button isLoading={processing.isLoading} onClick={onClick}>
    Click me
  </Button>
)
```
